Pull XKCD Comic page - JavaScript

const fetch = require('node-fetch');
module.exports = {
	name: 'comic',
	description: 'Gives a XKCD comic'
	args: false,
	execute: async(message, args) =>{
		if(!args.length || isNaN(args)){
			const {img} = await fetch(`https://xkcd.com/info.0.json`).then(response => response.json());
			message.channel.send(`Here is the latest XKCD comic: ${img}`);
		}
		else{
			const {img} = await fetch(`https://xkcd.com/${args}/info.0.json`).then(response => response.json());
			message.channel.send(`Here is XKCD comic #${args}: ${img}`);
		}
	}
};

const int CHUNK_SIZE = 512;

    int JPG_COUNTER = 0;
    uint8_t buffer[CHUNK_SIZE];
// Open the file that contains the images
    FILE *input = fopen(argv[1], "r");

// Loop through the data from the file, until it reaches the end
    while (fread(buffer, CHUNK_SIZE, 1, input))
    {
// Check if the first four bytes of a chunk match the header of a JPEG file
// Increment the JPG_COUNTER
// Apply the next number in sequence to the variable filename
// Write to a new JPEG file with the filename
// Close the JPEG file
        if (buffer[0] == 0xff && buffer[1] == 0xd8 && buffer[2] == 0xff && (buffer[3] & 0xf0) == 0xe0)
        {
            JPG_COUNTER++;
            sprintf(filename, "%03i.jpg", JPG_COUNTER);
            FILE *img = fopen(filename, "w");
            fwrite(&buffer, CHUNK_SIZE, 1, img);
            fclose(img);
        }

// Re-open the JPEG file
// Appened the next chunk of 512 bytes
// Close the JPEG file
        else
        {
            FILE *img = fopen(filename, "a");
            fwrite(&buffer, CHUNK_SIZE, 1, img);
            fclose(img);
        }
    }

    fclose(input);
}
